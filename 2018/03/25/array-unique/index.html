js数组去重

发表于 2018-03-25 | 分类于 js

「数组去重」是个老生常谈的问题了，这篇文章主要介绍了js数组去重的6种实现方法，感兴趣的小伙伴们可以参考一下。

for(两次) + 新数组

思路：
1.构建一个新的数组存放结果
2.for循环中每次从原数组中取出一个元素，用这个元素循环与结果数组对比
3.若结果数组中没有该元素，则存到结果数组中

let arr = [1, 'a', 'a', 'b', 'd', 'e', 'e', 1, 0, 2, 2, 3];
function unique(arr) {
    var newArr = [];
    for (var i = 1; i < arr.length; i++) {
        let flag = false;
        for (var j = 0; j < newArr.length; j++) {
            if (arr[i] == newArr[j]) {
                flag = true;
                break;
            }
        }
        if (!flag) {
            newArr.push(arr[i]);
        }
    }
    return newArr;
}

for(一次) + 新数组 + indexOf()

思路：
1.构建一个新的数组
2.如果当前数组的第i已经保存进了临时数组，那么跳过，否则把当前项push到临时数组里面

function unique(arr) {
    var newArr = [];  
    for (var i = 0; i < arr.length; i++) {
        if (newArr.indexOf(arr[i]) == -1){ 
            newArr.push(arr[i]);
        }
    }
    return newArr;
}

for(一次) + sort()排序 + 新数组

思路：
1.先将原数组进行排序
2.检查原数组中的第i个元素 与 结果数组中的最后一个元素是否相同，因为已经排序，所以重复元素会在相邻位置
3.如果不相同，则将该元素存入结果数组中

function unique4(arr) {
    arr.sort();
    var newArr = [];
    for (var i = 1; i < arr.length; i++) {
        if (arr[i] !== newArr[newArr.length - 1]) {
            newArr.push(arr[i]);
        }
    }
    return newArr;
}
